00107    public boolean validate(byte[] data, int off, int len) {
00108       for (int i = off; i < off + len; ++i) {
00109          mState = DFA[256 + (mState << 4) + DFA[(int) (0xff & data[i])]];
00110          if (mState == REJECT) {
00111             mPos += i;
00112             return false;
00113          }
00114       }
00115       mPos += len;
00116       return true;
